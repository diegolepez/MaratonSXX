<script>
const trivia = [
{ q: "¿En qué año ocurrió la caída del Muro de Berlín?", a: "1989", o: ["1985", "1991", "1980"] },
{ q: "¿Quién fue el primer hombre en pisar la Luna?", a: "Neil Armstrong", o: ["Buzz Aldrin", "Yuri Gagarin", "Michael Collins"] },
{ q: "¿En qué país comenzó la Revolución Industrial?", a: "Inglaterra", o: ["Francia", "EE.UU.", "Alemania"] },
{ q: "¿Qué evento inició la Primera Guerra Mundial?", a: "Asesinato del Archiduque Francisco Fernando", o: ["Invasión de Polonia", "Batalla de Verdún", "Hundimiento del Lusitania"] },
{ q: "¿En qué año se descubrió América?", a: "1492", o: ["1488", "1500", "1453"] }
];

const tilesData = [
    { x: 380, y: 520, n: &quot;1&quot;, c: &quot;t-yellow&quot; }, { x: 465, y: 525, n: &quot;2&quot;, c: &quot;t-orange&quot; },
    { x: 550, y: 525, n: &quot;3&quot;, c: &quot;t-pink&quot; }, { x: 635, y: 505, n: &quot;4&quot;, c: &quot;t-yellow&quot; },
    { x: 675, y: 440, n: &quot;5&quot;, c: &quot;t-orange&quot; }, { x: 600, y: 410, n: &quot;5&quot;, c: &quot;t-green&quot; },
    { x: 515, y: 405, n: &quot;&quot;, c: &quot;t-yellow&quot; }, { x: 430, y: 415, n: &quot;&quot;, c: &quot;t-orange&quot; },
    { x: 345, y: 435, n: &quot;6&quot;, c: &quot;t-yellow&quot; }, { x: 260, y: 435, n: &quot;&quot;, c: &quot;t-orange&quot; },
    { x: 175, y: 415, n: &quot;7&quot;, c: &quot;t-yellow&quot; }, { x: 175, y: 340, n: &quot;7&quot;, c: &quot;t-orange&quot; },
    { x: 250, y: 300, n: &quot;8&quot;, c: &quot;t-yellow&quot; }, { x: 335, y: 295, n: &quot;8&quot;, c: &quot;t-pink&quot; },
    { x: 420, y: 285, n: &quot;&quot;, c: &quot;t-green&quot; }, { x: 505, y: 265, n: &quot;&quot;, c: &quot;t-yellow&quot; },
    { x: 590, y: 245, n: &quot;&quot;, c: &quot;t-orange&quot; }, { x: 675, y: 220, n: &quot;8&quot;, c: &quot;t-yellow&quot; },
    { x: 755, y: 170, n: &quot;7&quot;, c: &quot;t-orange&quot; }, { x: 800, y: 100, n: &quot;&quot;, c: &quot;t-yellow&quot; }
];

let currentPos = 0, score = 0, time = 15, timer;

function initGame() {
    if(!document.getElementById(&#39;name-input&#39;).value) return;
    document.getElementById(&#39;start-screen&#39;).classList.add(&#39;hidden&#39;);
    document.getElementById(&#39;game-screen&#39;).classList.remove(&#39;hidden&#39;);
    renderTiles();
    moveAvatar(0);
    ask();
}

function renderTiles() {
    const layer = document.getElementById(&#39;tiles-layer&#39;);
    layer.innerHTML = &quot;&quot;;
    tilesData.forEach(t =&gt; {
        const d = document.createElement(&#39;div&#39;);
        d.className = `tile ${t.c}`;
        d.style.left = t.x + &quot;px&quot;;
        d.style.top = t.y + &quot;px&quot;;
        d.innerText = t.n;
        layer.appendChild(d);
    });
}

function moveAvatar(i) {
    const av = document.getElementById(&#39;avatar&#39;);
    av.style.left = (tilesData[i].x + 10) + &quot;px&quot;;
    av.style.top = (tilesData[i].y - 45) + &quot;px&quot;;
}

function startClock() {
    clearInterval(timer);
    time = 15;
    document.getElementById(&#39;ui-time&#39;).innerText = time;
    timer = setInterval(() =&gt; {
        time--;
        document.getElementById(&#39;ui-time&#39;).innerText = time;
        if(time &lt;= 0) { 
            clearInterval(timer); 
            alert(&quot;¡Tiempo agotado!&quot;); 
            location.reload(); 
        }
    }, 1000);
}

function ask() {
    const q = trivia[Math.floor(Math.random() * trivia.length)];
    document.getElementById(&#39;q-label&#39;).innerText = q.q;
    const container = document.getElementById(&#39;opt-container&#39;);
    container.innerHTML = &quot;&quot;;
    
    [...q.o, q.a].sort(() =&gt; Math.random() - 0.5).forEach(o =&gt; {
        const b = document.createElement(&#39;button&#39;);
        b.className = &quot;opt-btn&quot;;
        b.innerText = o;
        b.onclick = () =&gt; validate(o, q.a);
        container.appendChild(b);
    });
    
    document.getElementById(&#39;modal-trivia&#39;).classList.add(&#39;active&#39;);
    startClock();
}

function validate(s, c) {
    clearInterval(timer);
    document.getElementById(&#39;modal-trivia&#39;).classList.remove(&#39;active&#39;);
    
    const ok = s === c;
    const title = document.getElementById(&#39;f-title&#39;);
    const msg = document.getElementById(&#39;f-msg&#39;);
    const btn = document.getElementById(&#39;f-btn&#39;);

    if(ok) {
        score += 100;
        document.getElementById(&#39;ui-score&#39;).innerText = score;
        title.innerText = &quot;¡ÉXITO!&quot;;
        title.style.color = &quot;#16A34A&quot;;
        msg.innerText = &quot;¡Excelente respuesta! Sigue así.&quot;;
        btn.onclick = () =&gt; { 
            document.getElementById(&#39;modal-feedback&#39;).classList.remove(&#39;active&#39;); 
            advance(); 
        };
    } else {
        title.innerText = &quot;ERROR&quot;;
        title.style.color = &quot;#DC2626&quot;;
        msg.innerText = &quot;Ese dato no es correcto. Intenta de nuevo.&quot;;
        btn.onclick = () =&gt; { 
            document.getElementById(&#39;modal-feedback&#39;).classList.remove(&#39;active&#39;); 
            ask(); 
        };
    }
    document.getElementById(&#39;modal-feedback&#39;).classList.add(&#39;active&#39;);
}

function advance() {
    currentPos += 2;
    if(currentPos &gt;= tilesData.length) currentPos = tilesData.length - 1;
    moveAvatar(currentPos);
    
    if(currentPos === tilesData.length - 1) { 
        alert(&quot;¡META! Has completado la maratón. Puntos: &quot; + score); 
        location.reload(); 
    } else { 
        setTimeout(ask, 600); 
    }
}
